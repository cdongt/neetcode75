```python
class Solution(object):
    def search(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: int
        """
        if nums[len(nums)-1] < target:
            for i in range(len(nums)):
                if nums[i] == target:
                    return i
                if nums[i] < nums[i-1]:
                    break
            return -1
        else:
            for j in range(len(nums)-1,-1,-1):
                if nums[j] == target:
                    return j
                if  nums[j] <= nums[j-1]:
                    break
            return -1
```

```python
class Solution:
    def search(self, nums: List[int], target: int) -> int:
        l, r = 0, len(nums) - 1

        while l <= r:
            mid = (l + r) // 2
            if target == nums[mid]:
                return mid

            # left sorted portion
            if nums[l] <= nums[mid]:
                if target > nums[mid] or target < nums[l]:
                    l = mid + 1
                else:
                    r = mid - 1
            # right sorted portion
            else:
                if target < nums[mid] or target > nums[r]:
                    r = mid - 1
                else:
                    l = mid + 1
        return -1

```
