class Solution(object):
    def topKFrequent(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: List[int]
        """
        ndict = {}
        reslist = [[] for i in range(len(nums)+1)]
        output = []
        for n in nums:
            ndict[n] = 1 + ndict.get(n,0)
        for val , freq in ndict.items():
            reslist[freq].append(val)
        for i in range(len(reslist)-1,0,-1):
            for f in reslist[i]:
                output.append(f)
                if len(output) == k:
                    return output

class Solution(object):
    def topKFrequent(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: List[int]
        """
        rel = [[] for i in range(len(nums)+1)]
        ndict = {}
        output= []
        for val in nums:
            ndict[val] = 1 + ndict.get(val,0)
        
        for val,seq in ndict.items():
            rel[seq].append(val)
        
        for i in range(len(nums),0,-1):
            if rel[i] is None:
                continue
            for f in rel[i]:
                output.append(f)
                if len(output) == k:
                    return output
