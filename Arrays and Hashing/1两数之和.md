```
class Solution(object):
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        hashset = {}
        result =[]
        for i in range(len(nums)):
            if (target-nums[i]) in hashset:
                result.append(hashset[target-nums[i]])
                result.append(i)
                return result
            hashset[nums[i]] = i
        return None
```

```
            if diff in prevMap:
```
