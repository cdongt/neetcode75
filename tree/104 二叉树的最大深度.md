#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:

    def maxDepth(self, root):
        if not root:return 0
        res = 0
        sk = [[root,1]]
        while sk:
            node,count = sk.pop()
            res = max(res,count)
            if node.left: sk.append([node.left,count+1])
            if node.right: sk.append([node.right,count+1])
        return res
