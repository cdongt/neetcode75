from collections import deque
class Solution(object):
    def levelOrder(self, root):
        """
        :type root: TreeNode
        :rtype: List[List[int]]
        """
        if not root:return []
        dq = deque()
        dq.append(root)
        res = []
        while dq:
            vals = []
            for i  in range(len(dq)):
                node = dq.popleft()
                vals.append(node.val)
                if node.left: dq.append(node.left)
                if node.right: dq.append(node.right)
            res.append(vals)
        return res
